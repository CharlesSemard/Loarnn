86  	@Override
87  	public void doNothing() {
88  		
89  		if(isHit(this.getX(), this.getY())) {this.die();}
90  		else if(this.isOnSpell(this.getX(), this.getY())) 
91  			this.spell.collect();
92  		
93  		int index = 0;
94  		for(int i = 0; i < sprites.length; i++) {
95  			if(sprites[i] == this.getSprite()) {
96  				index = i;
97  			}
98  		}
99  		
100 		if(index < 7)
101 			this.setSprite(sprites[index + 1]);
102 		else
103 			this.setSprite(sprites[0]);
104 		this.setHasMoved();
105 	}

111 	public void specialCase(int x, int y) {
112 		if(isHit(this.getX(), this.getY())) {this.die();}
113 		else if(this.isOnKey(x, this.getY())) {}
114 		else if(this.isOnDoor(x, this.getY())) {}
115 		else if(this.isOnPurse(x, this.getY())) {}
116 		else if(this.isOnSpell(x, this.getY())) {this.spell.collect();}
117 	}

279 		if(gate.getX() == newX && gate.getY() == newY) {
280 			if(hasKey) {

305 		for(IMobile monster : this.monsters) {
306 			if(monster.getX() == newX && monster.getY() == newY && monster.isAlive()) {
307 				return true;

352 	 * Makes the spell spawn at a specific location
353 	 */
354 	public void shoot() {
355 		if(spell != null)
356 		if(!spell.isAlive()) {
357 			int direction = lastY != 0 ? (lastY == -1 ? 2 : 1) : (lastX == -1 ? 3 : 4);
358 			((Spell)this.spell).spawn(this.getX() - lastX, this.getY() - lastY, direction);
359 		}
360 	}
